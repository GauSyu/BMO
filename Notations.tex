\nomenclature{$[\Cc,\Dd], \Dd^{\Cc}$}{the category of functors from $\mathcal{C}$ to $\mathcal{D}$}
\nomenclature{$\mathcal{C}\simeq\mathcal{D}$}{$\mathcal{C}$ and $\mathcal{D}$ are equivalent}

\nomenclature{$\mathop{\underleftarrow{\lim}}D_j$}{the limit for $D$}
\nomenclature{$\mathop{\underrightarrow{\lim}}D_j$}{the colimit for $D$}
\nomenclature{$A\times_CB$}{the fibre product of $A$ and $B$ over $C$}

\nomenclature{$S^1$}{the unit circle}
\nomenclature{$\mathbf{Ring}$}{the category of rings}
\nomenclature{$\prod A_i$}{the product of $\{A_i\}_{i\in I}$}
\nomenclature{$\coprod A_i$}{the coproduct of $\{A_i\}_{i\in I}$}

\nomenclature{$\mathbf{Ban}_1$}{the category of Banach spaces and linear contractions}
